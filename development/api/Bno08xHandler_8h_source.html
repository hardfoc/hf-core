121constexpr const char* Bno08xErrorToString(Bno08xError error) noexcept {
122 switch (error) {
123 case Bno08xError::SUCCESS: return "Success";
124 case Bno08xError::NOT_INITIALIZED: return "Not initialized";
125 case Bno08xError::INITIALIZATION_FAILED: return "Initialization failed";
126 case Bno08xError::INVALID_PARAMETER: return "Invalid parameter";
127 case Bno08xError::COMMUNICATION_FAILED: return "Communication failed";
128 case Bno08xError::SENSOR_NOT_RESPONDING: return "Sensor not responding";
129 case Bno08xError::CALIBRATION_FAILED: return "Calibration failed";
130 case Bno08xError::FIRMWARE_UPDATE_FAILED: return "Firmware update failed";
131 case Bno08xError::TIMEOUT: return "Timeout";
132 case Bno08xError::MUTEX_LOCK_FAILED: return "Mutex lock failed";
133 case Bno08xError::INVALID_INTERFACE: return "Invalid interface";
134 case Bno08xError::SENSOR_NOT_ENABLED: return "Sensor not enabled";
135 case Bno08xError::DATA_NOT_AVAILABLE: return "Data not available";
136 case Bno08xError::HARDWARE_ERROR: return "Hardware error";
137 case Bno08xError::UNSUPPORTED_OPERATION: return "Unsupported operation";
138 default: return "Unknown error";
